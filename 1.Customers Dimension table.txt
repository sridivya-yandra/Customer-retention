CREATE OR REPLACE TABLE `Customer_Data.dim_customer` AS
WITH first_activity AS (
SELECT
customer_id,
MIN(from_date) AS acquisition_date
FROM `Customer_Data.Activity`
GROUP BY customer_id
)
SELECT
c.customer_id,
c.customer_country,
COALESCE(a.taxonomy_business_category_group, 'Unknown') AS acquisition_taxonomy,
fa.acquisition_date,
DATE_TRUNC(fa.acquisition_date, MONTH)      AS cohort_month
FROM `Customer_Data.Customer` c
LEFT JOIN `Customer_Data.Acq_orders` a  ON c.customer_id = a.customer_id
LEFT JOIN first_activity fa ON c.customer_id = fa.customer_id;
